Getting started

# Install Nextflow
curl -fsSL get.nextflow.io | bash
# Add Nextflow binary to your user's PATH:
mv nextflow ~/bin/

nextflow run nf-core/rnaseq -profile docker \
    --singleEnd --reads '*.fastq' --genome GRCm38

nextflow run nf-core/rnaseq -profile utd_ganymede \
    --singleEnd --reads '*.fastq' --genome GRCm38
